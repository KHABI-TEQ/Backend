  type: "banner" | "sidebar" | "popup" | "carousel" | "inline";
  startDate?: string; // ISO date string
  endDate?: string; // ISO date string
  isFeatured?: boolean;
  tags?: string[];
  status?: "active" | "inactive" | "expired" | "draft";
}

// 4. LIST PROMOTIONS
// GET /promotions/getAll?status=active&type=banner&search=summer&page=1&limit=20&startDate=2025-01-01&endDate=2025-12-31&isFeatured=true
interface ListPromotionsQuery {
  status?: "active" | "inactive" | "expired" | "draft";
  type?: "banner" | "sidebar" | "popup" | "carousel" | "inline";
  search?: string;
  page?: string;
  limit?: string;
  startDate?: string;
  endDate?: string;
  isFeatured?: string;
}

interface ListPromotionsResponse {
  success: true;
  message: "Promotions fetched successfully";
  data: Array<{
    _id: string;
    title: string;
    description?: string;
    imageUrl: string;
    redirectUrl?: string;
    type: string;
    startDate?: string;
    endDate?: string;
    isFeatured: boolean;
    tags: string[];
    views: number;
    clicks: number;
    status: string;
    createdBy: {
      _id: string;
      name: string;
      email: string;
    };
    createdAt: string;
    updatedAt: string;
  }>;
  total: number;
  page: number;
  limit: number;
  pages: number;
}

// 6. GET PROMOTION ANALYTICS
// GET /promotions/:id/analytics
interface GetPromotionAnalyticsResponse {
  success: true;
  message: "Promotion analytics fetched successfully";
  data: {
    promotionId: string;
    title: string;
    type: string;
    status: string;
    totalActivities: number;
    clicks: number;
    views: number;
    clickThroughRate: string; // e.g., "25.50%"
    uniqueUsers: number;
    topSources: Array<{
      ipAddress: string;
      activities: number;
    }>;
    createdAt: string;
    startDate?: string;
    endDate?: string;
  };
}
